# UserProfile

## Default Usage

The UserProfile component provides a popover-based user profile dropdown that displays user information and a logout button.

```tsx
import { UserProfile, Icon } from '@ssa-ui-kit/core';

const App = () => {
  const handleLogout = () => {
    console.log('User logged out');
    // Handle logout logic here
  };

  return (
    <UserProfile
      name="John Doe"
      email="john.doe@example.com"
      trigger={<Icon name="user" size={32} />}
      onClick={handleLogout}
    />
  );
};
```
